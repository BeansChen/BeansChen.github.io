# 代码实现

<!DOCTYPE html>
<html lang="zh">
	<head>
		<meta charset="UTF-8">
		<meta http-equiv="X-UA-Compatible" content="IE=edge">
		<meta name="viewport" content="width=device-width, initial-scale=1.0">
		<title>省市区联动</title>
		<link rel="stylesheet" href="https://res.wx.qq.com/open/libs/weui/2.0.1/weui.min.css">
	</head>
	<body>
		<script src="https://res.wx.qq.com/t/wx_fed/cdn_libs/res/weui/1.2.3/weui.min.js"></script>
		<script src="citydata.js"></script>
		<script type="text/javascript">
			// 级联picker
			var defaultValue = [cityData3[0].value,cityData3[0].children[0].value,cityData3[0].children[0].children[0].value]
			weui.picker(cityData3, {
			   defaultValue: defaultValue,
			   onChange: function (result) {
			       console.log(result)
			   },
			   onConfirm: function (result) {
			       console.log(result)
			   },
			});
		</script>
	</body>
</html>
