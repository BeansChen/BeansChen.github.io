开发小程序时遇到需要用户手写签名的需求，这里总结了新版和旧版canvas两种实现方式

# 新版

<template>
	<view>
		<view class="sign">
			<view class="sign__main">
				<view class="sign__canvas">
					<canvas
						v-if="showCanvas"
						type="2d"
						class="signCanvas"
						canvas-id="signCanvas"
						id="signCanvas"
						@touchstart="touchstart"
						@touchmove="touchmove"
						@touchend="touchend"
					></canvas>
				</view>
			</view>
			<view class="sign__bottom">
				<text class="sign__tips">请在上方空白区书写您的签字，字迹需工整</text>
				<view class="sign__clear" @click="handleClear">清除重写</view>
				<view class="sign__sub" @click="handleSubmit">提交签字</view>
			</view>
		</view>
	</view>
</template>

<script>
export default {
	data() {
		return {
			showCanvas: true,
			ctx: '', // 绘图对象
			points: [], //路径点集合
			originData: [],
		};
	},
	onLoad() {
		setTimeout(() => {
			this.createCanvas();
		}, 350);
	},
	methods: {
		// 将签名保存为本地图片
		handleSubmit() {
			if (this.originData.length === 0) {
				console.log('签字不能为空');
				return;
			}
			const query = uni
				.createSelectorQuery()
				// #ifndef MP-ALIPAY
				.in(this);
			// #endif
			query
				.select('#signCanvas')
				.fields({ node: true, size: true })
				.exec(res => {
					const canvasNode = res[0].node;
					uni.canvasToTempFilePath(
						{
							canvas: canvasNode,
							destWidth:654,	
							destHeight:368,
							success: res => {
								console.log(res)
							}
						},
						this
					);
				});
		},
		// 创建写字板
		createCanvas() {
			// 创建绘图对象
			const query = uni
				.createSelectorQuery()
				// #ifndef MP-ALIPAY
				.in(this);
			// #endif
			query
				.select('#signCanvas')
				.fields({
					node: true,
					size: true
				})
				.exec(res => {
					const canvasNode = res[0].node;

					const ctx = canvasNode.getContext('2d');
					const dpr = wx.getSystemInfoSync().pixelRatio;
					canvasNode.width = res[0].width * dpr;
					canvasNode.height = res[0].height * dpr;
					ctx.scale(dpr, dpr);

					ctx.fillStyle = '#ffffff';
					ctx.fillRect(0,0,canvasNode.width,canvasNode.height);
					ctx.strokeStyle = '#14161A';
					ctx.lineWidth = 6;
					ctx.lineCap = 'round';
					ctx.lineJoin = 'round';
					
					this.ctx = ctx;
					this.points = [];
					this.originData = [];
				});
		},
		touchstart(e) {
			const { points, originData, ctx } = this;
			const { x, y } = e.changedTouches[0];
			points.push([x, y]);
			originData.push([[x, y]]);
			ctx.moveTo(x, y);
		},
		touchmove(e) {
			const { points, originData, draw } = this;
			const { x, y } = e.changedTouches[0];
			points.push([x, y]);
			originData[originData.length - 1].push([x, y]);
			draw();
		},
		touchend(e) {
			this.points = [];
		},
		draw() {
			const { ctx, points } = this;
			if (points.length < 2) return;
			ctx.lineTo(...points[0]);
			ctx.stroke();
			ctx.moveTo(...points.shift());
		},
		// 清空画布
		handleClear() {
			this.showCanvas = false;
			this.$nextTick(() => {
				this.showCanvas = true;
				this.$nextTick(() => {
					this.createCanvas();
				});
			});
		},
		getRect(selector, all) {
			return new Promise(resolve => {
				uni
					.createSelectorQuery()
					.in(this)
					[all ? 'selectAll' : 'select'](selector)
					.boundingClientRect(rect => {
						if (all && Array.isArray(rect) && rect.length) {
							resolve(rect);
						}
						if (!all && rect) {
							resolve(rect);
						}
					})
					.exec();
			});
		}
	}
};
</script>

# 旧版

<template>
	<view>
		<view class="sign">
			<view class="sign__canvas">
				<canvas class="signCanvas" canvas-id="signCanvas" @touchstart="touchstart" @touchmove="touchmove" @touchend="touchend"></canvas>
			</view>
			<view class="sign__bottom">
				<text class="sign__tips">请在上方空白区书写您的签字，字迹需工整</text>
				<view class="sign__clear" @click="handleClear">清除重写</view>
				<view class="sign__sub" @click="handleSubmit">提交签字</view>
			</view>
		</view>
	</view>
</template>

<script>
export default {
	data() {
		return {
			ctx: '', // 绘图对象
			points: [], //路径点集合
			originData: []
		};
	},
	onLoad() {
		this.createCanvas();
	},
	methods: {
		// 创建写字板
		createCanvas() {
			// 创建绘图对象
			const ctx = uni.createCanvasContext('signCanvas', this);
			ctx.setStrokeStyle('#14161A');
			ctx.setLineWidth(6);
			ctx.setLineCap('round');
			ctx.setLineJoin('round');
			this.ctx = ctx;
			this.points = [];
			this.originData = [];
		},
		touchstart(e) {
			const { points, originData } = this;
			const { x, y } = e.changedTouches[0];
			points.push([x, y]);
			originData.push([[x, y]]);
		},
		touchmove(e) {
			const { points, originData, draw } = this;
			const { x, y } = e.changedTouches[0];
			points.push([x, y]);
			originData[originData.length - 1].push([x, y]);
			draw();
		},
		touchend(e) {
			this.points = [];
		},
		draw() {
			const { ctx, points } = this;
			if (points.length < 2) return;
			ctx.moveTo(...points.shift());
			ctx.lineTo(...points[0]);
			ctx.stroke();
			ctx.draw(true);
		},
		// 清空画布
		handleClear() {
			this.getRect('.signCanvas').then(rect => {
				this.ctx.clearRect(0, 0, rect.width, rect.height);
				this.ctx.draw(true);
				this.originData = [];
			});
		},
		getRect(selector, all) {
			return new Promise(resolve => {
				uni
					.createSelectorQuery()
					.in(this)
					[all ? 'selectAll' : 'select'](selector)
					.boundingClientRect(rect => {
						if (all && Array.isArray(rect) && rect.length) {
							resolve(rect);
						}
						if (!all && rect) {
							resolve(rect);
						}
					})
					.exec();
			});
		},
		// 将签名上传到服务器
		handleSubmit() {
			if(this.originData.length===0){
				console.log('签字不能为空')
				return
			}
			uni.canvasToTempFilePath(
				{
					canvasId: 'signCanvas',
					success(res){
						console.log(res)
					}
				},
				this
			)
		}
	}
};
</script>

# 总结

  1. 一开始用的旧版方式实现，但真机调试发现使用起来很卡，后来转而用新版方式来实现。
  2. 在ios端注意设置"disableScroll": true, 禁用ios的上拉下拉效果，否则签字版体验很差。
