form,fieldset,input,textarea,p,blockquote,th,td,i {
    padding: 0;
    margin: 0;
}

    box-sizing: border-box;
    -moz-box-sizing:border-box; /* Firefox */
    -webkit-box-sizing:border-box; /* Safari */ 
}
