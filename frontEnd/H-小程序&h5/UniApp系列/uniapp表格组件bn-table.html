# 前言

基于 uni-app 开发的小程序项目中有不少页面使用到表格，表格的基本功能样式都类似，但是有些表格会涉及到冻结单元格、冻结表头、分页、操作按钮等，于是决定把表格封装为一个公共组件。

# bn-table代码

<template>
	<view class="bn-table">
		<view v-if="showCaption" class="bn-table__caption" :class="[captionClassName]">
			<text>{{ captionText }}</text>
		</view>
		<scroll-view
			class="bn-table__scroll"
			scroll-y
			scroll-x
			:scroll-top="scrollTop"
			:scroll-left="scrollLeft"
			:style="[getTableStyle()]"
			@scroll="scroll"
			@scrolltolower="scrolltolower"
		>
			<view class="bn-table__container" :class="[border ? 'bn-table__container--border' : '']">
				<!-- 表头 -->
				<view class="bn-table__thead" v-if="showHeader">
					<view class="bn-table__tr">
						<view
							class="bn-table__cell"
							:class="[column.headerClassName, column.fixed ? 'bn-table__cell--fixed' : '']"
							:style="[column.headerCellStyle]"
							v-for="(column, cIndex) in transColumns"
							:key="cIndex"
						>
							<view class="bn-table__cell-text">
								<template v-if="column.type === 'selection'">
									<view class="bn-table__checkbox"><table-checkbox :indeterminate="indeterminate" :checked="checkedAll" @select="checkboxSelectedAll" /></view>
								</template>
								<template v-else>
									{{ column.label || '&nbsp;' }}
								</template>
							</view>
						</view>
					</view>
				</view>
				<!-- 主体 -->
				<view class="bn-table__tbody">
					<view
						class="bn-table__tr"
						:class="[
							stripe ? (index % 2 !== 0 ? 'bn-table__tr--striped' : '') : '',
							highlightCurrentRow && isHighlight(row, index) ? 'current-row' : '',
							rowClassName && getRowClassName(row, index)
						]"
						:style="[rowStyle && getRowStyle()]"
						@click.stop="rowClick(row, index)"
						v-for="(row, index) in transListData"
						:key="index"
					>
						<view
							class="bn-table__cell"
							:style="[column.cellStyle, cellStyle && getCellStyle(row, column, index, cIndex)]"
							:class="[column.className, column.fixed ? 'bn-table__cell--fixed' : '', cellClassName && getCellClassName(row, column, index, cIndex)]"
							v-for="(column, cIndex) in transColumns"
							:key="cIndex"
						>
							<view class="bn-table__cell-text">
								<template v-if="column.type === 'selection'">
									<view class="bn-table__checkbox"><table-checkbox @select="checkboxSelected($event, row)" :row-data="row" :checked="row.checked" /></view>
								</template>
								<template v-else-if="column.type === 'index'">
									{{ index + 1 }}
								</template>
								<template v-else-if="column.type === 'img'">
									<template v-if="row[column.key]">
										<image
											class="bn-table__img"
											@click.stop="preview(row[column.key], imgIndex)"
											:src="img"
											v-for="(img, imgIndex) in getImgs(row[column.key])"
											:key="imgIndex"
											mode="aspectFit"
										></image>
									</template>
									<template v-else>
										{{ emptyString }}
									</template>
									{{ lIndex + 1 }}
								</template>
								<template v-else-if="column.type === 'operate'">
									<view class="bn-table__operate">
										<template v-for="btn in column.operateList">
											<template v-if="btn.src">
												<image
													class="bn-table__btn2"
													@click.stop="handleOperate({ type: btn.type, row: row, index: index })"
													:src="btn.src"
													:key="btn.type"
													mode="aspectFit"
												></image>
											</template>
											<template v-else>
												<button
													class="bn-table__btn"
													:class="[btn.classes ? btn.classes : '']"
													@click.stop="handleOperate({ type: btn.type, row: row, index: index })"
													:key="btn.type"
												>
													{{ btn.label }}
												</button>
											</template>
										</template>
									</view>
								</template>
								<template v-else>
									{{ row[column.key] || emptyString }}
								</template>
							</view>
						</view>
					</view>
				</view>
				<!-- 底部-合计 -->
				<view class="bn-table__tfoot" v-if="showSummary">
					<view class="bn-table__tr">
						<view
							class="bn-table__cell"
							:style="[column.cellStyle]"
							:class="[column.className, column.fixed ? 'bn-table__cell--fixed' : '']"
							v-for="(column, cIndex) in transColumns"
							:key="cIndex"
						>
							<view class="bn-table__cell-text">{{ sums[cIndex] || emptyString }}</view>
						</view>
					</view>
				</view>
			</view>
			<!-- 上拉加载更多 -->
			<view style="width: 100%;" :style="{ 'margin-left': loadPadLeft + 'px' }"><table-load-more v-if="showLoadMore" :status="loadStatus"></table-load-more></view>
		</scroll-view>
	</view>
</template>

<script>
import TableCheckbox from './components/table-checkbox.vue';
import TableLoadMore from './components/table-load-more.vue';
export default {
	components: {
		TableCheckbox,
		TableLoadMore
	},
	props: {
		// 显示列
		columns: {
			type: Array,
			required: true
		},
		// 数据
		listData: {
			type: Array,
			required: true
		},
		// 斑马纹表格
		stripe: {
			type: Boolean,
			default: false
		},
		// 边框
		border: {
			type: Boolean,
			default: false
		},
		// 列的宽度是否自撑开
		fit: {
			type: Boolean,
			default: false
		},
		// Table 的高度，默认为自动高度。如果 height 为 number 类型，单位 rpx；如果 height 为 string 类型，则这个高度会设置为 Table 的 style.height 的值，Table 的高度会受控于外部样式。
		height: {
			type: [String, Number]
		},
		// Table 单元格字体大小
		fontSize: {
			type: Number,
			default: 28
		},
		// Table 单元格默认宽度
		cellWidth: {
			type: Number,
			default: 200
		},
		// 单元格对齐方式
		align: {
			type: String,
			default: 'left'
		},
		// 是否显示表头
		showHeader: {
			type: Boolean,
			default: true
		},
		// 行的 className 的回调方法，也可以使用字符串为所有行设置一个固定的 className
		rowClassName: {
			type: [String, Function]
		},
		// 行的 style 的回调方法，也可以使用一个固定的 Object 为所有行设置一样的 Style。
		rowStyle: {
			type: [Object, Function]
		},
		// 单元格的 className 的回调方法，也可以使用字符串为所有单元格设置一个固定的 className。
		cellClassName: {
			type: [String, Function]
		},
		// 单元格的 style 的回调方法，也可以使用一个固定的 Object 为所有单元格设置一样的 Style。
		cellStyle: {
			type: [Object, Function]
		},
		// 是否要高亮当前行
		highlightCurrentRow: {
			type: Boolean,
			default: false
		},
		// 是否在表尾显示合计行
		showSummary: {
			type: Boolean,
			default: false
		},
		// 合计行第一列的文本
		sumText: {
			type: String,
			default: '合计'
		},
		// 自定义的合计计算方法
		summaryMethod: {
			type: Function
		},
		// 显示标题
		showCaption: {
			type: Boolean,
			default: false
		},
		// 标题内容
		captionText: {
			type: String,
			default: '表格标题'
		},
		// 使用字符串为标题设置一个固定的 className
		captionClassName: {
			type: String,
			default: ''
		},
		// 行的 Key，用来优化Table的渲染
		rowKey: [String, Function],
		// 显示上拉加载更多
		showLoadMore: {
			type: Boolean,
			default: false
		},
		// 上拉的状态：more-loading前；loading-loading中；noMore-没有更多了
		loadStatus: {
			type: String,
			default: 'more'
		}
	},
	data() {
		return {
			transColumns: [],
			transListData: [],
			emptyString: '-',
			currentRowIndex: -1,
			currentRow: {},
			sums: [],
			selectedList: [],
			indeterminate: false,
			checkedAll: false,
			loadPadLeft: 0,
			scrollLeft: 0,
			scrollTop: 0,
			old: {
				scrollLeft: 0,
				scrollTop: 0
			}
		};
	},
	watch: {
		columnsProps: {
			deep: true,
			immediate: false,
			handler(newValue, oldValue) {
				this.initColumns(newValue);
			}
		},
		listDataProps: {
			deep: true,
			immediate: true,
			handler(newValue, oldValue) {
				this.initListData(newValue);
				this.initColumns(newValue);
				if (this.showSummary) {
					this.initSummary(newValue);
				}
			}
		},
		selectedList: {
			deep: true,
			immediate: false,
			handler(newValue, oldValue) {
				const selectedList = newValue;

				if (this.transListData.length) {
					// 表头选中按钮样式
					const len = this.transListData.filter(item => item.checked).length;
					if (len) {
						if (len === this.transListData.length) {
							this.checkedAll = true;
							this.indeterminate = false;
						} else {
							this.indeterminate = true;
						}
					} else {
						this.checkedAll = false;
						this.indeterminate = false;
					}
				} else {
					this.checkedAll = false;
					this.indeterminate = false;
				}
				this.$emit('selection-change', selectedList);
			}
		}
	},
	computed: {
		columnsProps() {
			return JSON.parse(JSON.stringify(this.columns));
		},
		listDataProps() {
			return JSON.parse(JSON.stringify(this.listData));
		}
	},
	methods: {
		initColumns() {
			let { columnsProps, cellWidth, fontSize, align, fit, listData } = this;
			columnsProps = JSON.parse(JSON.stringify(columnsProps));
			this.transColumns = columnsProps.map((column, index) => {
				// 表头样式
				column.headerCellStyle = column.headerCellStyle || {};
				// 主体样式
				column.cellStyle = column.cellStyle || {};
				// 单元格宽度自动撑开
				if (fit && !column.width) {
					if (column.type === 'selection') {
						column.width = 120;
					} else if (column.type === 'index') {
						column.width = 120;
					} else if (column.type === 'img') {
						let arr = [this.getTextWidth(column.label)];
						listData.forEach(data => {
							if (data[column.key]) {
								let urls = this.getImgs(data[column.key]);
								arr.push(urls.length * 74); // 每张图片占据的空间 74
							}
						});
						column.width = Math.max(...arr) + 36;
					} else if (column.type === 'operate') {
						let arr = [this.getTextWidth(column.label)];
						if (column.operateList) {
							let w = column.operateList.reduce((pre, cur) => {
								// 每个按钮占据的空间 字数*28 + 30
								return pre + (cur.src ? 64 : cur.label.length * 28 + 30);
							}, 0);
							arr.push(w);
						}
						column.width = Math.max(...arr) + 32;
					} else {
						let arr = [this.getTextWidth(column.label)];
						listData.forEach(data => {
							if (data[column.key]) {
								let w = this.getTextWidth(data[column.key]);
								arr.push(w);
							}
						});
						column.width = Math.max(...arr) + 36;
					}
				}
				// 宽度
				if (typeof column.width !== 'number') {
					column.width = cellWidth;
				}
				// 公共样式
				const commonStyle = {
					fontSize: fontSize + 'rpx',
					textAlign: align,
					width: column.width + 'rpx'
				};

				// 左右固定
				if (column.fixed) {
					if (column.fixed === 'right') {
						const nextColumns = columnsProps.slice(index + 1);
						const right = nextColumns.reduce((pre, cur) => {
							if (typeof cur.width !== 'number') {
								return pre + cellWidth;
							}
							return pre + cur.width;
						}, 0);
						commonStyle.right = right + 'rpx';
					} else {
						const preColumns = columnsProps.slice(0, index);
						const left = preColumns.reduce((pre, cur) => {
							if (typeof cur.width !== 'number') {
								return pre + cellWidth;
							}
							return pre + cur.width;
						}, 0);
						commonStyle.left = left + 'rpx';
					}
				}
				column.headerCellStyle = {
					...column.headerCellStyle,
					...commonStyle,
					textAlign: column.headerAlign || column.align || 'left'
				};
				column.cellStyle = {
					...column.cellStyle,
					...commonStyle,
					textAlign: column.align || 'left'
				};
				return column;
			});
		},
		initListData(listDataProps) {
			const selectionColumn = this.columnsProps.find(item => item.type === 'selection');
			const selectedList = [];
			const listData = listDataProps.map((item, index) => {
				if (this.rowKey) {
					if (typeof this.rowKey === 'function') {
						item.key = Object.freeze(this.rowKey(item)) || Date.now();
					} else {
						item.key = Object.freeze(item[this.rowKey]) || Date.now();
					}
				} else {
					item.key = index;
				}
				// 数据更新后保留之前选中的数据（翻页或者搜索后）
				if (selectionColumn) {
					if (selectionColumn.reserveSelection) {
						if (item.checked === true) {
							selectedList.push(item);
						} else {
							const i = this.selectedList.findIndex(v => v.key === item.key);
							if (i > -1) {
								item.checked = true;
								selectedList.push(item);
							} else {
								item.checked = false;
							}
						}
					} else {
						item.checked = false;
					}
				}
				return item;
			});
			this.transListData = listData;
			this.selectedList = selectedList;
		},
		initSummary(listDataProps) {
			const {} = this;
			let sums = [];
			if (this.summaryMethod) {
				sums = this.summaryMethod({ columns: this.columnsProps, data: listDataProps });
			} else {
				this.columnsProps.forEach((column, index) => {
					if (index === 0) {
						sums[index] = this.sumText;
						return;
					}
					const values = listDataProps.map(item => Number(item[column.key]));
					const precisions = [];
					let notNumber = true;
					values.forEach(value => {
						if (!isNaN(value)) {
							notNumber = false;
							let decimal = ('' + value).split('.')[1];
							precisions.push(decimal ? decimal.length : 0);
						}
					});
					const precision = Math.max.apply(null, precisions);
					if (!notNumber) {
						sums[index] = values.reduce((pre, cur) => {
							const value = Number(cur);
							if (!isNaN(value)) {
								return parseFloat((pre + cur).toFixed(Math.min(precision, 20)));
							} else {
								return pre;
							}
						}, 0);
					} else {
						sums[index] = '';
					}
				});
			}
			this.sums = sums;
		},
		checkboxSelectedAll(e, row) {
			if (e.checked) {
				const selectedList = [];
				this.transListData.forEach(item => {
					item.checked = true;
					selectedList.push(item);
				});
				this.selectedList = selectedList;
			} else {
				this.transListData.forEach(item => {
					item.checked = false;
				});
				this.selectedList = [];
			}
			this.$forceUpdate();
		},
		checkboxSelected(e, row) {
			this.transListData.forEach(item => {
				if (item.key === row.key) {
					item.checked = e.checked;
				}
			});
			if (e.checked) {
				this.selectedList.push(row);
			} else {
				this.selectedList = this.selectedList.filter(item => item.key !== row.key);
			}
			this.$forceUpdate();
			this.$emit('select', this.selectedList, row);
		},
		isHighlight(row, rowIndex) {
			return rowIndex === this.currentRowIndex;
		},
		getTableStyle() {
			let { height } = this;
			const tableStyle = {};
			if (height) {
				tableStyle.height = typeof height === 'number' ? height + 'rpx' : height;
			}
			return tableStyle;
		},
		getRowClassName(row, rowIndex) {
			const rowClassName = this.rowClassName;
			if (typeof rowClassName === 'function') {
				return rowClassName({ row, rowIndex });
			}
			return rowClassName || '';
		},
		getRowStyle(row, rowIndex) {
			const rowStyle = this.rowStyle;
			if (typeof rowStyle === 'function') {
				return rowStyle({ row, rowIndex });
			}
			return rowStyle || {};
		},
		getCellClassName(row, column, rowIndex, columnIndex) {
			const cellClassName = this.cellClassName;
			if (typeof cellClassName === 'function') {
				return cellClassName({ row, column, rowIndex, columnIndex });
			}
			return cellClassName || '';
		},
		getCellStyle(row, column, rowIndex, columnIndex) {
			let cellStyle = this.cellStyle || {};
			if (typeof cellStyle === 'function') {
				cellStyle = cellStyle({ row, column, rowIndex, columnIndex });
			}
			return cellStyle || {};
		},
		getImgs(imgs) {
			if (typeof imgs === 'string') {
				return imgs.split(',');
			} else if (Array.isArray(imgs)) {
				return imgs;
			} else {
				return [];
			}
		},
		// 计算文字所占宽度
		getTextWidth(str) {
			str = str + '';
			const { fontSize } = this;
			let regx = /^[0-9]+.?[0-9]*$/;
			let flexWidth = 0;
			for (const char of str) {
				if ((char >= 'A' && char <= 'Z') || (char >= 'a' && char <= 'z')) {
					// 如果是英文字符，为字符分配0.58单位宽度
					flexWidth += Math.ceil(fontSize * 0.58);
				} else if (char >= '\u4e00' && char <= '\u9fa5') {
					// 如果是中文字符，为字符分配1.08单位宽度
					flexWidth += Math.ceil(fontSize * 1.08);
				} else if (regx.test(char)) {
					// 如果是数字，为字符分配0.6单位宽度
					flexWidth += Math.ceil(fontSize * 0.6);
				} else {
					// 其他种类字符，为字符分配0.58个单位宽度
					flexWidth += Math.ceil(fontSize * 0.58);
				}
			}
			return flexWidth < 64 ? 64 : flexWidth;
		},
		rowClick(row, index) {
			if (this.highlightCurrentRow) {
				this.currentRowIndex = index;
				this.currentRow = row;
				this.$emit('currentChange', { row, index });
			}
			this.$emit('rowClick', { row, index });
		},
		preview(imgs, current) {
			if (typeof imgs === 'string') {
				imgs = imgs.split(',');
			}
			if (!Array.isArray(imgs)) {
				return;
			}
			uni.previewImage({
				current,
				urls: imgs
			});
		},
		handleOperate(data) {
			this.$emit('opreate', { detail: data });
		},
		scroll(e) {
			const { scrollLeft, scrollTop } = e.detail;
			this.loadPadLeft = Math.floor(scrollLeft);
			this.old.scrollLeft = scrollLeft;
			this.old.scrollTop = scrollTop;
		},
		scrolltolower(e) {
			const direction = e.detail.direction;
			if (direction == 'bottom') {
				this.$emit('scrollBottom');
			}
		},
		/**
		 * 外部调用方法
		 * ===================================================
		 */
		// 返回顶部
		goTop() {
			this.scrollTop = this.old.scrollTop;
			this.$nextTick(function() {
				this.scrollTop = 0;
			});
		},
		// 返回左侧
		goLeft() {
			this.scrollLeft = this.old.scrollLeft;
			this.$nextTick(function() {
				this.scrollLeft = 0;
			});
		}
	}
};
</script>

<script>
export default {
	props: {
		// 是否半选中
		indeterminate: {
			type: Boolean,
			default: false
		},
		// 是否选中
		checked: {
			type: [Boolean, String],
			default: false
		},
		// 是否禁用
		disabled: {
			type: Boolean,
			default: false
		}
	},
	watch: {
		checked: {
			deep: true,
			immediate: true,
			handler(newValue, oldValue) {
				if (typeof this.checked === 'boolean') {
					this.isChecked = newValue;
				} else {
					this.isChecked = true;
				}
			}
		},
		indeterminate: {
			deep: true,
			immediate: true,
			handler(newValue, oldValue) {
				this.isIndeterminate = newValue;
			}
		}
	},
	data() {
		return {
			isChecked: false,
			isIndeterminate: false
		};
	},
	created() {},
	methods: {
		selected() {
			if (this.disabled) return;
			this.isIndeterminate = false;
			this.isChecked = !this.isChecked;
			console.log('===', this.indeterminate, this.isChecked);
			this.$emit('select', {
				checked: this.isChecked
			});
		}
	}
};
</script>
