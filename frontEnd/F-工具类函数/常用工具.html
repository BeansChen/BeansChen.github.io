# 时间格式化

/**
 * 格式化时间
 * @method formatDate
 * @param {string} date 时间
 * @param {string} fmt 匹配模式 yyyy-MM-dd hh:mm:ss
 * return {string} 2069-03-23 14:26:28
 * @author 张三 2050/07/29
 * @version 1.0.0
 * @example formatDate(new Date(),'yyyy-MM-dd')
 */
const formatDate = (date, fmt) => {
	if(!date){
		return ''
	}
	if (typeof(date) == "string" && date.toString().indexOf("-") != -1) {
		date = date.replace(/\-/g, '/')
	}
	var date = new Date(date)
	if (/(y+)/.test(fmt)) {
		fmt = fmt.replace(RegExp.$1, (date.getFullYear() + '').substr(4 - RegExp.$1.length));
	}
	let o = {
		'M+': date.getMonth() + 1,
		'd+': date.getDate(),
		'h+': date.getHours(),
		'm+': date.getMinutes(),
		's+': date.getSeconds()
	};
	for (let k in o) {
		if (new RegExp(`(${k})`).test(fmt)) {
			let str = o[k] + '';
			fmt = fmt.replace(RegExp.$1, (RegExp.$1.length === 1) ? str : padLeftZero(str));
		}
	}

	function padLeftZero(str) {
		return ('00' + str).substr(str.length);
	}
	return fmt;
}

# 删除富文本标签，只保留文本

/**
 * 删除html标签，只保留文本
 * @description 用于根据富文本内容提取出缩略简介
 * @method delHtmlTags
 * @param {string} val 富文本
 * @return {string} 纯文本字符串
 * @author 张三 2050/07/29
 * @version 1.0.0
 */
export const delHtmlTags = (val) => {
	val = val || '';
	return val.replace(/<[^>]+>/g, "");
}

# 数值千分位格式化，并保留n位小数

/**
 * 数值千分位格式化，并保留n位小数
 * @method formatThousands
 * @param {number|string} val 数值
 * @param {number} num 保留小数位数 0-20
 * @return {string} 1,000.00
 * @author 张三 2050/07/29
 * @version 1.0.0
 */
const formatThousands = (val, num = 0) => {
	val = parseFloat(val.toString()).toFixed(num).toString();
	if (isNaN(val)) return '';
	const arr = val.split('.');
	const int = arr[0].split('');
	const decimal = arr[1] || '';
	let temp = ''
	int.reverse().forEach(function(v, i) {
		if (i !== 0 && i % 3 === 0 && v !== '-') {
			temp = v + "," + temp;
		} else {
			temp = v + temp;
		}
	})
	return temp + (!!decimal ? "." + decimal : '');
}

export const getName = (val) => {
  val = val || "";
  return val.replace(/\s*/g, "").substring(val.length - 2);
};

# 从一个数组随机取出n项

/**
 * 从一个数组随机取出n项
 * @method getRandomArr
 * @param {Array} arr 原数组
 * @param {Number} num 随机数组长度
 * @return {Array} 随机数组
 * @author 张三 2050/07/29
 * @version 1.0.0
 */

 const getRandomArr = (arr, num) => {
  const length = arr.length;
  if (num > length) return arr;
  const tempArr = arr.slice(0)
  const res = []
  for (let i = 0; i < num; i++) {
    const random = Math.floor(Math.random() * tempArr.length)
    res.push(tempArr.splice(random, 1)[0])
  }
  return res
}

function GetRequest() {
	var url = location.search; //获取url中"?"符后的字串
	var theRequest = new Object();
	if (url.indexOf("?") != -1) {
		var str = url.substr(1);
		strs = str.split("&");
		for (var i = 0; i < strs.length; i++) {
			theRequest[strs[i].split("=")[0]] = unescape(strs[i].split("=")[1]);
		}
	}
	return theRequest;
}

# 绘制圆环

<canvas width="200" height="200" class="circle"></canvas>
<script>
  /**
			 * 绘制圆环
			 */
  let drawRingProgress = function(obj) {
    var options = {
      ele: obj.ele,
      percent: obj.percent, //进度
      forecolor: obj.forecolor || '#3ab3eb', //圆环前景色
      bgcolor: obj.bgcolor || '#ebf7fd', //圆环背景色
      text: obj.text || '进度', //圆环描述文字
      lineWidth: obj.lineWidth || 20,
    }
    var drawing_elem = document.querySelector(options.ele);
    var context = drawing_elem.getContext("2d");
    var center_x = drawing_elem.width / 2; //横坐标中间点
    var center_y = drawing_elem.height / 2; //纵坐标中间点
    console.log(center_x, center_y)
    var rad = Math.PI * 2 / 100;
    var speed = 0;

    // 绘制背景圆圈  
    function backgroundCircle() {
      context.save();
      context.beginPath();
      context.lineWidth = options.lineWidth; //设置线宽  
      var radius = center_x - context.lineWidth;
      context.lineCap = "round";
      context.strokeStyle = options.bgcolor;
      context.arc(center_x, center_y, radius, 0, Math.PI * 2, false);
      context.stroke();
      context.closePath();
      context.restore();
    }

    //绘制运动圆环  
    function foregroundCircle(n) {
      context.save();
      context.strokeStyle = options.forecolor;
      context.lineWidth = options.lineWidth;
      context.lineCap = "round";
      var radius = center_x - context.lineWidth;
      context.beginPath();
      context.arc(center_x, center_y, radius, -Math.PI / 2, -Math.PI / 2 + n * rad,
                  false); //用于绘制圆弧context.arc(x坐标，y坐标，半径，起始角度，终止角度，顺时针/逆时针)  
      context.stroke();
      context.closePath();
      context.restore();
    }

    //绘制进度文字  
    function drawSpeed(n) {
      context.save(); //save和restore可以保证样式属性只运用于该段canvas元素  
      context.fillStyle = options.forecolor;
      var font_size = Math.ceil(30);
      context.font = font_size + "px Helvetica";
      var text_width = context.measureText(n + "%").width;
      context.fillText(n + "%", center_x - text_width / 2, center_y - 2);
      context.restore();
    }
    //绘制文字  
    function drawText(n) {
      context.save(); //save和restore可以保证样式属性只运用于该段canvas元素  
      context.fillStyle = "#999";
      var font_size = Math.ceil(20);
      context.font = font_size + "px Helvetica";
      var text_width = context.measureText(n).width;
      context.fillText(n, center_x - text_width / 2, center_y + font_size + 2);
      context.restore();
    }

    //执行动画  
    (function drawFrame() {
      window.requestAnimationFrame(drawFrame);
      context.clearRect(0, 0, drawing_elem.width, drawing_elem.height);
      backgroundCircle();
      drawSpeed(speed);
      drawText(options.text);
      foregroundCircle(speed);
      if (options.percent - speed > 0 && options.percent - speed < 1) {
        speed = parseFloat(options.percent);
      }
      if (options.percent - speed >= 1) {
        speed += 1;
      }
      if (options.percent - speed < 0) {
        return;
      }
    }());
  }
  drawRingProgress({
    ele: '.circle',
    percent: 20, //进度
  })
</script>

# Rem

module.exports = function() {
	var doc = document;
	//设计稿分辨率
	var rWidth = arguments[0] || 750;
	//设置视口
	var vMeta = doc.getElementById("viewport");
	vMeta.name = "viewport";
	vMeta.content = "width=device-width,initial-scale=1,minimum-scale=1,maximum-scale=1,user-scalable=no";
	//设置缩放值
	var docEl = doc.documentElement,
		resizeEvt = 'orientationchange' in window ? 'orientationchange' : 'resize',
		recalc = function() {
			var clientWidth = docEl.clientWidth;
			if(!clientWidth) return;
			clientWidth = (clientWidth > rWidth) ? rWidth : clientWidth;
			docEl.style.fontSize = 100 * (clientWidth / rWidth) + 'px';
			window.REM = 100 * (clientWidth / rWidth);
		};
	if(!doc.addEventListener) return;
	window.addEventListener(resizeEvt, recalc, false);
	recalc();
};
