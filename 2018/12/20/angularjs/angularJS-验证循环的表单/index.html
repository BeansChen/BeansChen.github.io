angularJS验证ng-repeat循环出的表单

前言

在使用angularJS与后端做交互时，都会用它的表单验证功能，但有时候表单属性并不是一个对象，而是一个对象列表，比如需要提交一个不定长度的联系人列表。
我在项目遇到这个情况的时候，第一时间就直接对form作了循环，遗憾的是angular并不能获取到form对象，中途试了很多种方法，都不尽人意，最后中总结了两种
实现的方法：

使用ng-form

angularjs的指令ng-form是支持嵌套的表单的，可以使用ng-form实现。

<!DOCTYPE html>
<html lang="en" ng-app="app">

<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<meta http-equiv="X-UA-Compatible" content="ie=edge">
<title>Document</title>
<script src="http://cdn.bootcss.com/angular.js/1.5.6/angular.js"></script>
</head>

<body>
<div ng-controller="formController">
    <button type="button" ng-click="addContactObj()">新增</button>
    <form name="outerForm" novalidate>
    <ng-form name="innerForm" ng-repeat="item in contactList">
        <h2>联系人{{$index}} <button type="button" ng-click="delContactObj($index,contactList)">删除</button></h2>
        <div>
        <div><input type="text" ng-model="item.name" name="username" placeholder="请输入姓名" ng-pattern="/^[\u4e00-\u9fff]{2,6}$/"
            required></div>
        <div style="color:red;" ng-show="innerForm.username.$error.pattern && !innerForm.username.$untouched">请输入2~6位中文</div>
        <div style="color:red;" ng-show="innerForm.username.$error.required && !innerForm.username.$untouched">姓名不能为空</div>
        </div>
        <div>
        <div><input type="text" maxlength="3" ng-model="item.age" name="age" placeholder="请输入年龄" required></div>
        <div style="color:red;" ng-show="innerForm.age.$error.required && !innerForm.age.$untouched">年龄不能为空</div>
        </div>
    </ng-form>

    </form>
    <button type="button" ng-click="sub()">提交</button>
</div>
<script>
    var app = angular.module('app', []);
    app.controller('formController', function ($scope) {
    $scope.contactList = [];
    $scope.addContactObj = function () {
        var contactObj = {
        name: "",
        age: ""
        }
        $scope.contactList.push(contactObj);
    }
    $scope.delContactObj = function (key, arr) {
        arr.splice(key, 1);
    }
    $scope.sub = function () {
        if(!validForm()) return;
        alert("提交成功！");
    }
    function validForm(){
        var error = $scope.outerForm.$error;
        if (error && error.required) {
        for (var i in error.required) {
            for(var j in error.required[i].$error.required){
            error.required[i].$error.required[j].$touched = true;
            error.required[i].$error.required[j].$untouched = false;
            return false;
            }
        }
        } else if (error.pattern) {
        for (var i in error.pattern) {
            for(var j in error.pattern[i].$error.pattern){
            error.pattern[i].$error.pattern[j].$touched = true;
            error.pattern[i].$error.pattern[j].$untouched = false;
            return false;
            }
        }
        } else if (error.maxlength) {
        for (var i in error.maxlength) {
            for(var j in error.maxlength[i].$error.maxlength){
            error.maxlength[i].$error.maxlength[j].$touched = true;
            error.maxlength[i].$error.maxlength[j].$untouched = false;
            return false;
            }
        }
        }
        if(!$scope.outerForm.$valid)return false;
        return true;      
    }
    });
</script>
</body>

</html>

使用循环的div

form元素不能循环，但是div元素却可以，再使用[]的方式取得对应的表单元素作验证

<!DOCTYPE html>
<html lang="en" ng-app="app">

<head>
<meta charset="UTF-8">
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<meta http-equiv="X-UA-Compatible" content="ie=edge">
<title>Document</title>
<script src="http://cdn.bootcss.com/angular.js/1.5.6/angular.js"></script>
</head>

<body>
<div ng-controller="formController">
    <button type="button" ng-click="addContactObj()">新增</button>
    <form name="outerForm" novalidate>
    <div ng-repeat="item in contactList">
        <h2>联系人{{$index}} <button type="button" ng-click="delContactObj($index,contactList)">删除</button></h2>
        <div>
        <div><input type="text" ng-model="item.name" name="{{'username'+$index}}" placeholder="请输入姓名" ng-pattern="/^[\u4e00-\u9fff]{2,6}$/"
            required></div>
        <div style="color:red;" ng-show="outerForm['username'+$index].$error.pattern && !outerForm['username'+$index].$untouched">请输入2~6位中文</div>
        <div style="color:red;" ng-show="outerForm['username'+$index].$error.required && !outerForm['username'+$index].$untouched">姓名不能为空</div>
        </div>
        <div>
        <div><input type="text" maxlength="3" ng-model="item.age" name="{{'age'+$index}}" placeholder="请输入年龄" required></div>
        <div style="color:red;" ng-show="outerForm['age'+$index].$error.required && !outerForm['age'+$index].$untouched">年龄不能为空</div>
        </div>
    </div>

    </form>
    <button type="button" ng-click="sub()">提交</button>
</div>

    var app = angular.module('app', []);
    app.controller('formController', function ($scope) {
    $scope.contactList = [];
    $scope.addContactObj = function () {

        name: "",

        }

    }

        arr.splice(key, 1);
    }
    $scope.sub = function () {
        if(!validForm()) return;
        alert("提交成功！");
    }
    function validForm(){
        var error = $scope.outerForm.$error;
        if (error && error.required) {
        for (var i in error.required) {
            error.required[i].$touched = true;
            error.required[i].$untouched = false;
            return false;
        }
        } else if (error.pattern) {
        for (var i in error.pattern) {
            error.pattern[i].$touched = true;
            error.pattern[i].$untouched = false;
            return false;
        }
        } else if (error.maxlength) {
        for (var i in error.maxlength) {
            error.maxlength[i].$touched = true;
            error.maxlength[i].$untouched = false;
            return false;
        }
        }
        if(!$scope.outerForm.$valid)return false;
        return true;      
    }
    });
</script>
</body>

</html>
